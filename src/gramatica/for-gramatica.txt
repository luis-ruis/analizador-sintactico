S 		      = for(EXP_INI; EXP_COND; EXP_BUCLE) { SENTENCIA }
EXP_INI 	  = IDENTIFICADOR = VALOR COMA__EXP_INI
VALOR 		  = IDENTIFICADOR | NUMERO
COMA__EXP_INI 	= , EXP_INI | ⋲
EXP_COND	  = OP_NEG VALOR OP_LOG | VALOR OP_REL VALOR OP_LOG
OP_NEG		  = ! | ⋲
OP_REL		  = < | > | <= | >= | == | !=
OP_LOG 		  = || EXP_COND | && EXP_COND | ⋲
EXP_BUCLE   = IDENTIFICADOR INC COMA_EXP_BUCLE
INC         = ++ | --
COMA_EXP_BUCLE = , EXP_BUCLE | ⋲
IDENTIFICADOR = LETRAS NUMEROS LETRAS
LETRAS      =  LETRA | ⋲
LETRA       = a...z LETRAS | A...Z LETRAS
NUMEROS     = NUMERO | ⋲
NUMERO      = 0...9 NUMEROS
SENTENCIAS = SENTENCIA | ⋲
SENTENCIA = IDENTIFICADOR = LADO_DERECHO SENTENCIAS ;
LADO_DERECHO = VALOR | VALOR OP_ARITMETICO VALOR MAS_LADO_DERECHO | (VALOR) | (VALOR OP_ARITMETICO VALOR MAS_LADO_DERECHO)
MAS_LADO_DERECHO = LADO_DERECHO | ⋲
OP_ARITMETICO = + | - | * | /
